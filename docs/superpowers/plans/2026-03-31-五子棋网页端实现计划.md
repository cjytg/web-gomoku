```typescript
// src/utils/gameLogic.ts
import { BOARD_SIZE, EMPTY, PLAYER1, PLAYER2, SCORES, DIFFICULTY } from '../constants/game';
import type { Board, Player, Position, Difficulty as DifficultyType } from '../types';

// 创建空棋盘
export const createEmptyBoard = (): Board => {
  return Array(BOARD_SIZE).fill(null).map(() => Array(BOARD_SIZE).fill(EMPTY));
};

// 检查位置是否有效
export const isValidPosition = (row: number, col: number): boolean => {
  return row >= 0 && row < BOARD_SIZE && col >= 0 && col < BOARD_SIZE;
};

// 检查胜负
export const checkWin = (board: Board, row: number, col: number, player: Player): boolean => {
  const directions = [[0, 1], [1, 0], [1, 1], [1, -1]];
  
  for (const [dr, dc] of directions) {
    let count = 1;
    
    // 正方向
    for (let i = 1; i < 5; i++) {
      const nr = row + dr * i;
      const nc = col + dc * i;
      if (isValidPosition(nr, nc) && board[nr][nc] === player) {
        count++;
      } else {
        break;
      }
    }
    
    // 反方向
    for (let i = 1; i < 5; i++) {
      const nr = row - dr * i;
      const nc = col - dc * i;
      if (isValidPosition(nr, nc) && board[nr][nc] === player) {
        count++;
      } else {
        break;
      }
    }
    
    if (count >= 5) {
      return true;
    }
  }
  
  return false;
};

// 检查棋盘是否已满
export const isBoardFull = (board: Board): boolean => {
  return board.every(row => row.every(cell => cell !== EMPTY));
};
```

- [ ] **Step 2: 实现AI算法**
```typescript
// 检查周围是否有邻居
const hasNeighbor = (board: Board, row: number, col: number, distance: number = 1): boolean => {
  for (let dr = -distance; dr <= distance; dr++) {
    for (let dc = -distance; dc <= distance; dc++) {
      if (dr === 0 && dc === 0) continue;
      const nr = row + dr;
      const nc = col + dc;
      if (isValidPosition(nr, nc) && board[nr][nc] !== EMPTY) {
        return true;
      }
    }
  }
  return false;
};

// 获取棋型分数
const getPatternScore = (count: number, block: number, empty: number): number => {
  if (count >= 5) return SCORES.FIVE;
  if (block === 2) return 0;
  if (count === 4) {
    return block === 0 ? SCORES.LIVE_FOUR : SCORES.DEAD_FOUR;
  } else if (count === 3) {
    if (block === 0) {
      return SCORES.LIVE_THREE + (empty >= 2 ? 10 : 0);
    } else {
      return SCORES.DEAD_THREE;
    }
  } else if (count === 2) {
    if (block === 0) {
      return SCORES.LIVE_TWO + (empty >= 2 ? 5 : 0);
    } else {
      return SCORES.DEAD_TWO;
    }
  } else if (count === 1) {
    return SCORES.ONE;
  }
  return 0;
};

// 计算位置分数
const calculateScore = (board: Board, row: number, col: number, player: Player): number => {
  let score = 0;
  const directions = [[0, 1], [1, 0], [1, 1], [1, -1]];
  
  for (const [dr, dc] of directions) {
    let count = 1;
    let block = 0;
    let empty = 0;
    
    // 正方向
    for (let i = 1; i < 5; i++) {
      const nr = row + dr * i;
      const nc = col + dc * i;
      if (!isValidPosition(nr, nc)) {
        block++;
        break;
      }
      if (board[nr][nc] === player) {
        count++;
      } else if (board[nr][nc] === EMPTY) {
        empty++;
        break;
      } else {
        block++;
        break;
      }
    }
    
    // 反方向
    for (let i = 1; i < 5; i++) {
      const nr = row - dr * i;
      const nc = col - dc * i;
      if (!isValidPosition(nr, nc)) {
        block++;
        break;
      }
      if (board[nr][nc] === player) {
        count++;
      } else if (board[nr][nc] === EMPTY) {
        empty++;
        break;
      } else {
        block++;
        break;
      }
    }
    
    score += getPatternScore(count, block, empty);
  }
  
  return score;
};

// 评估位置
const evaluatePosition = (board: Board, row: number, col: number, aiPlayer: Player, difficulty: DifficultyType): number => {
  const humanPlayer = aiPlayer === PLAYER2 ? PLAYER1 : PLAYER2;
  const aiScore = calculateScore(board, row, col, aiPlayer);
  const humanScore = calculateScore(board, row, col, humanPlayer);
  const config = DIFFICULTY[difficulty];
  
  if (aiScore >= SCORES.FIVE) return aiScore * 2;
  if (humanScore >= SCORES.FIVE) return humanScore * 1.5;
  
  return aiScore + humanScore * config.defenseWeight;
};

// 获取AI最佳落子
export const getAIMove = (board: Board, aiPlayer: Player, difficulty: DifficultyType = 'medium'): Position => {
  let bestScore = -1;
  let bestMoves: Position[] = [];
  const config = DIFFICULTY[difficulty];
  
  // 只考虑有邻居的位置
  for (let row = 0; row < BOARD_SIZE; row++) {
    for (let col = 0; col < BOARD_SIZE; col++) {
      if (board[row][col] === EMPTY && hasNeighbor(board, row, col)) {
        const score = evaluatePosition(board, row, col, aiPlayer, difficulty);
        if (score > bestScore) {
          bestScore = score;
          bestMoves = [{ row, col }];
        } else if (score === bestScore) {
          bestMoves.push({ row, col });
        }
      }
    }
  }
  
  // 棋盘为空时走中心
  if (bestMoves.length === 0) {
    const center = Math.floor(BOARD_SIZE / 2);
    return { row: center, col: center };
  }
  
  // 根据难度添加随机性
  if (Math.random() < config.randomness && bestMoves.length > 1) {
    // 寻找次优解
    let secondBestScore = -1;
    let secondBestMoves: Position[] = [];
    
    for (let row = 0; row < BOARD_SIZE; row++) {
      for (let col = 0; col < BOARD_SIZE; col++) {
        if (board[row][col] === EMPTY && 
            hasNeighbor(board, row, col) && 
            !bestMoves.some(move => move.row === row && move.col === col)) {
          const score = evaluatePosition(board, row, col, aiPlayer, difficulty);
          if (score > secondBestScore) {
            secondBestScore = score;
            secondBestMoves = [{ row, col }];
          } else if (score === secondBestScore) {
            secondBestMoves.push({ row, col });
          }
        }
      }
    }
    
    if (secondBestMoves.length > 0) {
      return secondBestMoves[Math.floor(Math.random() * secondBestMoves.length)];
    }
  }
  
  return bestMoves[Math.floor(Math.random() * bestMoves.length)];
};
```

```typescript
// src/hooks/useGame.ts
import { useState, useCallback } from 'react';
import { createEmptyBoard, checkWin, isBoardFull, getAIMove } from '../utils/gameLogic';
import { PLAYER1, PLAYER2, EMPTY } from '../constants/game';
import type { Board, Player, Position, GameMode, Difficulty } from '../types';

interface UseGameProps {
  mode: GameMode;
  difficulty?: Difficulty;
  player?: Player;
}

export const useGame = ({ mode, difficulty = 'medium', player = PLAYER1 }: UseGameProps) => {
  const [board, setBoard] = useState<Board>(createEmptyBoard());
  const [currentPlayer, setCurrentPlayer] = useState<Player>(PLAYER1);
  const [gameOver, setGameOver] = useState(false);
  const [winner, setWinner] = useState<Player | null>(null);
  const [lastMove, setLastMove] = useState<Position | null>(null);
  const [moveHistory, setMoveHistory] = useState<Position[]>([]);

  // 落子
  const makeMove = useCallback((row: number, col: number, player?: Player): boolean => {
    if (gameOver || board[row][col] !== EMPTY) return false;
    
    const playerToPlace = player || currentPlayer;
    const newBoard = board.map(r => [...r]);
    newBoard[row][col] = playerToPlace;
    
    setBoard(newBoard);
    setLastMove({ row, col });
    setMoveHistory(prev => [...prev, { row, col }]);
    
    // 检查胜负
    if (checkWin(newBoard, row, col, playerToPlace)) {
      setGameOver(true);
      setWinner(playerToPlace);
      return true;
    }
    
    // 检查平局
    if (isBoardFull(newBoard)) {
      setGameOver(true);
      setWinner(null);
      return true;
    }
    
    // 切换玩家
    if (!player) {
      setCurrentPlayer(currentPlayer === PLAYER1 ? PLAYER2 : PLAYER1);
    }
    
    // AI模式下自动走棋
    if (mode === 'ai' && !player && currentPlayer === PLAYER1) {
      setTimeout(() => {
        const aiMove = getAIMove(newBoard, PLAYER2, difficulty);
        makeMove(aiMove.row, aiMove.col, PLAYER2);
      }, 300);
    }
    
    return true;
  }, [board, currentPlayer, gameOver, mode, difficulty]);

  // 悔棋
  const undoMove = useCallback(() => {
    if (moveHistory.length === 0) return;
    
    const newHistory = [...moveHistory];
    let movesToUndo = mode === 'ai' ? 2 : 1;
    
    for (let i = 0; i < movesToUndo && newHistory.length > 0; i++) {
      newHistory.pop();
    }
    
    const newBoard = createEmptyBoard();
    let last: Position | null = null;
    
    newHistory.forEach((move, index) => {
      newBoard[move.row][move.col] = index % 2 === 0 ? PLAYER1 : PLAYER2;
      last = move;
    });
    
    setBoard(newBoard);
    setMoveHistory(newHistory);
    setLastMove(last);
    setGameOver(false);
    setWinner(null);
    setCurrentPlayer(newHistory.length % 2 === 0 ? PLAYER1 : PLAYER2);
  }, [moveHistory, mode]);

  // 重新开始
  const restartGame = useCallback(() => {
    setBoard(createEmptyBoard());
    setCurrentPlayer(PLAYER1);
    setGameOver(false);
    setWinner(null);
    setLastMove(null);
    setMoveHistory([]);
  }, []);

  return {
    board,
    currentPlayer,
    gameOver,
    winner,
    lastMove,
    makeMove,
    undoMove,
    restartGame,
    setBoard,
    setCurrentPlayer,
    setGameOver,
    setWinner,
    setLastMove,
  };
};
```

```typescript
// src/hooks/useMultiplayer.ts
import { useState, useEffect, useCallback } from 'react';
import { supabase } from '../supabaseClient';
import type { Room, Player, Position } from '../types';
import { PLAYER1, PLAYER2 } from '../constants/game';

export const useMultiplayer = () => {
  const [room, setRoom] = useState<Room | null>(null);
  const [playerId, setPlayerId] = useState<string>('');
  const [isHost, setIsHost] = useState(false);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  // 生成6位随机房间号
  const generateRoomCode = () => {
    return Math.floor(100000 + Math.random() * 900000).toString();
  };

  // 创建房间
  const createRoom = useCallback(async () => {
    setLoading(true);
    setError(null);
    
    try {
      const code = generateRoomCode();
      const id = crypto.randomUUID();
      setPlayerId(id);
      setIsHost(true);

      const { data, error } = await supabase
        .from('rooms')
        .insert([
          {
            code,
            player1: id,
            currentPlayer: PLAYER1,
            board: JSON.stringify(Array(15).fill(null).map(() => Array(15).fill(0))),
            status: 'waiting',
          }
        ])
        .select()
        .single();

      if (error) throw error;
      
      const newRoom = { ...data, board: JSON.parse(data.board) };
      setRoom(newRoom);
      
      // 订阅房间变化
      subscribeToRoom(newRoom.id);
      
      return newRoom;
    } catch (err: any) {
      setError(err.message);
      return null;
    } finally {
      setLoading(false);
    }
  }, []);

  // 加入房间
  const joinRoom = useCallback(async (code: string) => {
    setLoading(true);
    setError(null);
    
    try {
      // 查找房间
      const { data: rooms, error: findError } = await supabase
        .from('rooms')
        .select('*')
        .eq('code', code)
        .eq('status', 'waiting');
      
      if (findError) throw findError;
      if (rooms.length === 0) throw new Error('房间不存在或已开始游戏');
      
      const room = rooms[0];
      const id = crypto.randomUUID();
      setPlayerId(id);
      setIsHost(false);

      // 更新房间信息，加入第二个玩家
      const { data: updatedRoom, error: updateError } = await supabase
        .from('rooms')
        .update({
          player2: id,
          status: 'playing',
        })
        .eq('id', room.id)
        .select()
        .single();

      if (updateError) throw updateError;
      
      const parsedRoom = { ...updatedRoom, board: JSON.parse(updatedRoom.board) };
      setRoom(parsedRoom);
      
      // 订阅房间变化
      subscribeToRoom(room.id);
      
      return parsedRoom;
    } catch (err: any) {
      setError(err.message);
      return null;
    } finally {
      setLoading(false);
    }
  }, []);

  // 订阅房间实时变化
  const subscribeToRoom = useCallback((roomId: string) => {
    supabase
      .channel(`room:${roomId}`)
      .on('postgres_changes', {
        event: 'UPDATE',
        schema: 'public',
        table: 'rooms',
        filter: `id=eq.${roomId}`,
      }, (payload) => {
        const updatedRoom = payload.new as any;
        setRoom({ ...updatedRoom, board: JSON.parse(updatedRoom.board) });
      })
      .subscribe();
  }, []);

  // 落子同步
  const makeMove = useCallback(async (row: number, col: number, player: Player) => {
    if (!room) return;
    
    try {
      const newBoard = [...JSON.parse(JSON.stringify(room.board))];
      newBoard[row][col] = player;
      
      const { error } = await supabase
        .from('rooms')
        .update({
          board: JSON.stringify(newBoard),
          currentPlayer: player === PLAYER1 ? PLAYER2 : PLAYER1,
          last_move: JSON.stringify({ row, col }),
        })
        .eq('id', room.id);
      
      if (error) throw error;
    } catch (err: any) {
      setError(err.message);
    }
  }, [room]);

  // 结束游戏
  const endGame = useCallback(async (winner?: Player) => {
    if (!room) return;
    
    try {
      const { error } = await supabase
        .from('rooms')
        .update({
          status: 'finished',
          winner,
        })
        .eq('id', room.id);
      
      if (error) throw error;
    } catch (err: any) {
      setError(err.message);
    }
  }, [room]);

  // 重新开始游戏
  const restartGame = useCallback(async () => {
    if (!room) return;
    
    try {
      const { error } = await supabase
        .from('rooms')
        .update({
          board: JSON.stringify(Array(15).fill(null).map(() => Array(15).fill(0))),
          currentPlayer: PLAYER1,
          last_move: null,
          status: 'playing',
          winner: null,
        })
        .eq('id', room.id);
      
      if (error) throw error;
    } catch (err: any) {
      setError(err.message);
    }
  }, [room]);

  return {
    room,
    playerId,
    isHost,
    loading,
    error,
    createRoom,
    joinRoom,
    makeMove,
    endGame,
    restartGame,
  };
};
```
